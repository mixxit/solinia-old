FILE(GLOB SRC *.cpp *.h)

ADD_EXECUTABLE(file_info ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(file_info ${LIBXML2_LIBRARIES} ${PLATFORM_LINKFLAGS} nel3d)
IF(WIN32)
  SET_TARGET_PROPERTIES(file_info PROPERTIES 
    LINK_FLAGS "/NODEFAULTLIB:libcmt"
    PROJECT_LABEL "Tools, 3D: file_info")
ENDIF(WIN32)
ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS file_info RUNTIME DESTINATION bin COMPONENT tools3d)
